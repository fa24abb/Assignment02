# histogram of the data
plt.figure(figsize=(10, 6))  # Set figure size for better readability
plt.hist(data, bins=20, color='skyblue', edgecolor='black', alpha=0.7)

# Mean and Median
mean = np.mean(data)
median = np.median(data)

# annotations for Mean
plt.axvline(mean, color='red', linestyle='dashed', linewidth=2, label=f'Mean: {mean:.2f}')
plt.text(mean + 0.05, 25, f'Mean: {mean:.2f}', color='red', fontsize=12)

# annotations for Median
plt.axvline(median, color='green', linestyle='dashed', linewidth=2, label=f'Median: {median:.2f}')
plt.text(median - 0.2, 25, f'Median: {median:.2f}', color='green', fontsize=12)

# appearance for readability
plt.title("Histogram of Uniform Distribution Data", fontsize=16)
plt.xlabel("Value", fontsize=14)
plt.ylabel("Frequency", fontsize=14)
plt.legend(loc='upper right', fontsize=12)
plt.tick_params(axis='both', labelsize=12)

# Show the plot
plt.tight_layout()
plt.show()
